{assign 'array' ['ky', 'sh', 'ch','ny','hy','my','ry','gy','j','by','py']} {foreach $array AS $g }
{char_col c=$g|cat:'a'} {char_col c=$g|cat:'u' boxcolor='blue'} {char_col c=$g|cat:'o' boxcolor='violet'}
{if $g=='j'}
{char_col c=$g|cat:'a2' c_view="(ja)"} {char_col c=$g|cat:'u2' boxcolor='blue' c_view="(ju)"} {char_col c=$g|cat:'o2' boxcolor='violet' c_view="(jo)"}
{/if} {/foreach}
